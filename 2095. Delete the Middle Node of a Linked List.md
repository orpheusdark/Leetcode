class Solution {
public:
    ListNode* deleteMiddle(ListNode* head) {
        
        if (!head || !head->next){
            // delete head;
            return NULL;
        }

        // initialize slow at head 
        //and fast 2-step ahead to skip 1-move of slow to reach before middle
        ListNode* slow = head;
        ListNode* fast = head->next->next;

        while(fast != NULL && fast -> next != NULL){

            slow = slow -> next;
            fast = fast -> next -> next;
        }

        ListNode* middle = slow -> next;
        slow -> next = slow -> next ->next;
        delete middle;

        return head;
    }
};
